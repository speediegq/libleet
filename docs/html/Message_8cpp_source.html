9void leet::sendMessage(leet::User::CredentialsResponse* resp, leet::Room::Room* room, leet::Message::Message* msg) {
10 using json = nlohmann::json;
11 const int TransID { leet::TransID };
12 const std::string eventType { "m.room.message" };
13 const std::string APIUrl { "/_matrix/client/v3/rooms/" + room->roomID + "/send/" + eventType + "/" + std::to_string(TransID) };
14
15 json list;
16
17 if (!msg->messageType.compare("m.image") || !msg->messageType.compare("m.audio") || !msg->messageType.compare("m.video") || !msg->messageType.compare("m.file")) {
18 if (msg->attachmentURL[0] != 'm' || msg->attachmentURL[1] != 'x' || msg->attachmentURL[2] != 'c') {
19 leet::errorCode = 1;
20 return;
21 }
22
23 list["type"] = "m.room.message";
24 list["room_id"] = room->roomID;
25 list["body"] = msg->messageText;
26 list["msgtype"] = msg->messageType;
27 list["url"] = msg->attachmentURL;
28 } else {
29 list["type"] = "m.room.message";
30 list["room_id"] = room->roomID;
31 list["body"] = msg->messageText;
32 list["msgtype"] = msg->messageType;
33 }
34
35 const std::string Output { leet::invokeRequest_Put(leet::getAPI(APIUrl), list.dump(), resp->accessToken) };
36
37 json reqOutput;
38 try {
39 reqOutput = { json::parse(Output) };
40 } catch (const json::parse_error& e) {
41 return;
42 }
43
44 for (auto& output : reqOutput) {
45 leet::errorCode = 0;
46
47 if (output["errcode"].is_string()) {
48 leet::errorCode = 1;
49 leet::Error = output["errcode"].get<std::string>();
50 if (output["error"].is_string()) leet::friendlyError = output["error"].get<std::string>();
51 }
52 }
53}

90const std::vector<leet::Message::Message> leet::returnMessages(leet::User::CredentialsResponse* resp, leet::Room::Room* room, const int messageCount) {
91 using json = nlohmann::json;
92 std::vector<leet::Message::Message> vector;
93 const std::string APIUrl { "/_matrix/client/v3/rooms/" + room->roomID + "/messages?dir=b&limit=" + std::to_string(messageCount) };
94
95 std::string Output { leet::invokeRequest_Get(leet::getAPI(APIUrl), resp->accessToken) };
96
97 json reqOutput;
98 try {
99 reqOutput = json::parse(Output);
100 } catch (const json::parse_error& e) {
101 return vector;
102 }
103
104 auto& messages = reqOutput["chunk"];
105
106 for (auto it = messages.begin(); it != messages.end(); ++it) {
107 leet::Message::Message message;
108
109 message.Encrypted = false;
110
111 if (it.value().contains("/type"_json_pointer)) message.Type = it.value()["type"];
112
113 // Encrypted message
114 if (!message.Type.compare("m.room.encrypted")) {
115 message.Encrypted = true;
116 message.megolm = false;
117
118 if (it.value().contains("/content/ciphertext"_json_pointer)) message.cipherText = it.value()["content"]["ciphertext"];
119 if (it.value().contains("/content/sender_key"_json_pointer)) message.senderKey = it.value()["content"]["sender_key"];
120 if (it.value().contains("/content/device_id"_json_pointer)) message.deviceID = it.value()["content"]["device_id"];
121 if (it.value().contains("/content/session_id"_json_pointer)) message.sessionID = it.value()["content"]["session_id"];
122 if (it.value().contains("/content/algorithm"_json_pointer)) if (it.value()["content"]["algorithm"] == "m.megolm.v1.aes-sha2") message.megolm = true;
123 }
124
125 if (it.value().contains("/content/msgtype"_json_pointer)) message.messageType = it.value()["content"]["msgtype"];
126 if (it.value().contains("/sender"_json_pointer)) message.Sender = it.value()["sender"];
127
128 if (it.value().contains("/content/body"_json_pointer)) message.messageText = it.value()["content"]["body"];
129 if (it.value().contains("/content/formatted_body"_json_pointer)) message.formattedText = it.value()["content"]["formatted_body"];
130 if (it.value().contains("/content/format"_json_pointer)) message.Format = it.value()["content"]["format"];
131 if (it.value().contains("/content/info/mimetype"_json_pointer)) message.mimeType = it.value()["content"]["info"]["mimetype"];
132 if (it.value().contains("/event_id"_json_pointer)) message.eventID = it.value()["event_id"];
133 if (it.value().contains("/origin_server_ts"_json_pointer)) message.Age = it.value()["origin_server_ts"];
134
135 // Attachments
136 if (it.value().contains("/content/info/size"_json_pointer)) message.attachmentSize = it.value()["content"]["info"]["size"];
137 if (it.value().contains("/content/info/duration"_json_pointer)) message.attachmentLength = it.value()["content"]["info"]["duration"];
138 if (it.value().contains("/content/info/w"_json_pointer)) message.attachmentWidth = it.value()["content"]["info"]["w"];
139 if (it.value().contains("/content/info/h"_json_pointer)) message.attachmentHeight = it.value()["content"]["info"]["h"];
140 if (it.value().contains("/content/url"_json_pointer)) message.attachmentURL = it.value()["content"]["url"];
141
142 // Handle thumbnails
143 if (!message.messageType.compare("m.video")) {
144 if (it.value().contains("/content/info/thumbnail_info/w"_json_pointer)) message.thumbnailWidth = it.value()["content"]["info"]["thumbnail_info"]["w"];
145 if (it.value().contains("/content/info/thumbnail_info/h"_json_pointer)) message.thumbnailHeight = it.value()["content"]["info"]["thumbnail_info"]["h"];
146 if (it.value().contains("/content/info/thumbnail_info/size"_json_pointer)) message.thumbnailSize = it.value()["content"]["info"]["thumbnail_info"]["size"];
147 if (it.value().contains("/content/info/thumbnail_info/mimetype"_json_pointer)) message.thumbnailMimeType = it.value()["content"]["info"]["thumbnail_info"]["mimetype"];
148 if (it.value().contains("/content/info/thumbnail_url"_json_pointer)) message.thumbnailURL = it.value()["content"]["info"]["thumbnail_url"];
149 }
150
151 vector.push_back(message);
152 }
153
154 return vector;
155}

157const std::string leet::returnFilter(leet::User::CredentialsResponse* resp, leet::Filter::Filter *filter) {
158 using json = nlohmann::json;
159 const std::string APIUrl { "/_matrix/client/v3/user/" + resp->userID + "/filter" };
160
161 json list;
162
163 list["event_format"] = "client";
164 list["event_fields"] = filter->Fields;
165 list["presence"]["senders"] = filter->Senders;
166 list["presence"]["not_senders"] = filter->notSenders;
167 list["room"]["ephemeral"]["rooms"] = filter->Rooms;
168 list["room"]["ephemeral"]["not_rooms"] = filter->notRooms;
169 list["room"]["ephemeral"]["senders"] = filter->Senders;
170 list["room"]["ephemeral"]["not_senders"] = filter->notSenders;
171 list["room"]["state"]["rooms"] = filter->Rooms;
172 list["room"]["state"]["not_rooms"] = filter->notRooms;
173
174 if (filter->Limit != 0) {
175 list["room"]["timeline"]["limit"] = filter->Limit;
176 }
177
178 list["room"]["timeline"]["not_rooms"] = filter->notRooms;
179 list["room"]["timeline"]["not_senders"] = filter->notSenders;
180
181 std::string Output { leet::invokeRequest_Post(leet::getAPI(APIUrl), list.dump(), resp->accessToken) };
182
183 json reqOutput;
184 try {
185 reqOutput = json::parse(Output);
186 } catch (const json::parse_error& e) {
187 return "";
188 }
189
190 for (auto& output : reqOutput) {
191 leet::errorCode = 0;
192
193 if (output["filter_id"].is_string()) {
194 return output["filter_id"].get<std::string>();
195 }
196 }
197
198 return "";
199}
