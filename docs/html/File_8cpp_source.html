56leet::Attachment::Attachment leet::uploadFile(leet::User::CredentialsResponse* resp, const std::string& File) {
57 using json = nlohmann::json;
58 leet::Attachment::Attachment theAttachment;
59 const std::string Output = leet::invokeRequest_Post_File(leet::getAPI("/_matrix/media/v3/upload"), File, resp->accessToken);
60
61 json returnOutput;
62 try {
63 returnOutput = { json::parse(Output) };
64 } catch (const json::parse_error& e) {
65 return theAttachment;
66 }
67
68 for (auto& output : returnOutput) {
69 leet::errorCode = 0;
70 leet::Error = "";
71
72 if (output["content_uri"].is_string()) {
73 theAttachment.URL = output["content_uri"].get<std::string>();
74 return theAttachment;
75 }
76
77 if (output["errcode"].is_string()) {
78 leet::errorCode = 1;
79 leet::Error = output["errcode"].get<std::string>();
80 if (output["error"].is_string()) leet::friendlyError = output["error"].get<std::string>();
81 return theAttachment;
82 }
83 }
84
85 return theAttachment;
86}

88const bool leet::downloadFile(leet::User::CredentialsResponse* resp, leet::Attachment::Attachment* Attachment, const std::string& outputFile) {
89 std::string Server{};
90 std::string ID{};
91 std::string File{Attachment->URL};
92 std::size_t it = File.find("mxc://");
93
94 if (it != std::string::npos) {
95 it += 6;
96 std::size_t nextSlash = File.find("/", it);
97
98 if (nextSlash != std::string::npos) {
99 Server = File.substr(it, nextSlash - it);
100 ID = File.substr(nextSlash + 1);
101 } else {
102 leet::errorCode = 1;
103 return false;
104 }
105 }
106
107 // Now that we have what we need, let's make a request
108 const std::string API { leet::getAPI("/_matrix/media/v3/download/" + Server + "/" + ID + "?allow_redirect=false") };
109 std::filesystem::path file{ outputFile };
110
111 if (!std::filesystem::create_directories(file.parent_path()) && !std::filesystem::is_directory(file.parent_path())) {
112 leet::errorCode = 1;
113 return false;
114 }
115
116 // Download the file
117 std::ofstream of(outputFile, std::ios::binary);
118 cpr::Response response = cpr::Download(of, cpr::Url{API});
119 if (response.status_code == 200) {
120 return true;
121 }
122
123 return false;
124}
