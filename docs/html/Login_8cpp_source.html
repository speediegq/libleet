9std::vector<std::string> leet::returnSupportedLoginTypes() {
10 using json = nlohmann::json;
11 std::vector<std::string> vector;
12 const std::string APIUrl { "/_matrix/client/v3/login" };
13
14 std::string Output { leet::invokeRequest_Get(leet::getAPI(APIUrl)) };
15
16 json reqOutput;
17 try {
18 reqOutput = json::parse(Output);
19 } catch (const json::parse_error& e) {
20 return vector;
21 }
22
23 auto& messages = reqOutput["flows"];
24
25 for (auto it = messages.begin(); it != messages.end(); ++it) {
26 std::string theString{};
27 if (it.value().contains("/type"_json_pointer)) theString = it.value()["type"];
28 vector.push_back(theString);
29 }
30
31 return vector;
32}

34leet::User::CredentialsResponse leet::refreshAccessToken(leet::User::CredentialsResponse* resp) {
35 using json = nlohmann::json;
36 leet::User::CredentialsResponse newResponse = *resp;
37
38 if (!newResponse.refreshToken.compare("")) {
39 return newResponse;
40 }
41
42 json body;
43
44 body["refresh_token"] = newResponse.refreshToken;
45
46 json refreshOutput;
47
48 try {
49 refreshOutput = { json::parse(leet::invokeRequest_Post(leet::getAPI("/_matrix/client/v3/refresh"), body.dump())) };
50 } catch (const json::parse_error& e) {
51 return newResponse;
52 }
53
54 for (auto& output : refreshOutput) {
55 leet::errorCode = 0;
56
57 if (output["access_token"].is_string()) newResponse.accessToken = output["access_token"].get<std::string>();
58 if (output["refresh_token"].is_string()) newResponse.refreshToken = output["refresh_token"].get<std::string>();
59 if (output["expires_in_ms"].is_number_integer()) newResponse.Expiration = output["expires_in_ms"].get<int>();
60 if (output["errcode"].is_string()) {
61 leet::errorCode = 1;
62 leet::Error = output["errcode"].get<std::string>();
63 if (output["error"].is_string()) leet::friendlyError = output["error"].get<std::string>();
64 }
65 }
66
67 return newResponse;
68}

99leet::User::CredentialsResponse leet::registerAccount(leet::User::Credentials* cred) {
100 leet::User::CredentialsResponse resp;
101 using json = nlohmann::json;
102
103 std::string theUsername = cred->Username;
104
105 if (cred->Username[0] == '@') {
106 theUsername = leet::returnUserName(cred->Username);
107
108 if (theUsername[0] == '@' || !theUsername.compare("")) {
109 return resp;
110 leet::errorCode = 1;
111 }
112 }
113
114 json body;
115
116 if (cred->deviceID.compare("")) {
117 body["device_id"] = cred->deviceID;
118 }
119
120 body["inhibit_login"] = false;
121 body["initial_device_display_name"] = cred->displayName;
122 body["username"] = theUsername;
123 body["password"] = cred->Password;
124 body["refresh_token"] = cred->refreshToken;
125
126 json registerOutput;
127
128 try {
129 registerOutput = { json::parse(leet::invokeRequest_Post(leet::getAPI("/_matrix/client/v3/register"), body.dump())) };
130 } catch (const json::parse_error& e) {
131 return resp;
132 }
133
134 for (auto& output : registerOutput) {
135 leet::errorCode = 0;
136
137 resp.Homeserver = leet::Homeserver = cred->Homeserver;
138
139 if (output["access_token"].is_string()) resp.accessToken = output["access_token"].get<std::string>();
140 if (output["device_id"].is_string()) resp.deviceID = output["device_id"].get<std::string>();
141 if (output["refresh_token"].is_string()) resp.refreshToken = output["refresh_token"].get<std::string>();
142 if (output["user_id"].is_string()) resp.userID = output["user_id"].get<std::string>();

156 leet::User::CredentialsResponse resp;
157 using json = nlohmann::json;
158 json list;
159
160 std::string actualType{};
161
162 if (cred->Type == LEET_TYPE_TOKEN) {
163 actualType = "m.login.token";
164 } else {
165 actualType = "m.login.password";
166 }
167
168 if (cred->deviceID.compare("")) {
169 list["device_id"] = cred->deviceID;
170 }
171
172 list["identifier"]["type"] = "m.id.user"; // Currently only supported method
173 list["identifier"]["user"] = cred->Username;
174 list["initial_device_display_name"] = cred->displayName;
175
176 if (cred->Type == LEET_TYPE_TOKEN) {
177 list["token"] = cred->Token;
178 } else {
179 list["password"] = cred->Password;
180 }
181
182 list["refresh_token"] = cred->refreshToken;
183 list["type"] = actualType;
184
185 json loginOutput = { json::parse(leet::invokeRequest_Post(leet::getAPI("/_matrix/client/v3/login"), list.dump())) };
186
187 for (auto& output : loginOutput) {
188 leet::errorCode = 0;
189
190 resp.Homeserver = leet::Homeserver = cred->Homeserver;
191
192 if (output["access_token"].is_string()) resp.accessToken = output["access_token"].get<std::string>();
193 if (output["device_id"].is_string()) resp.deviceID = output["device_id"].get<std::string>();
194 if (output["refresh_token"].is_string()) resp.refreshToken = output["refresh_token"].get<std::string>();

197
198 if (output["errcode"].is_string()) {
