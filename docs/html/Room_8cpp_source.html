9/* Converts an alias to a proper room ID */
10const std::string leet::findRoomID(std::string Alias) {
11 using json = nlohmann::json;
12
13 leet::errorCode = 0;
14
15 if (Alias[0] == '!') { // It's a proper room ID already
16 return Alias;
17 }
18
19 // Replace the '#' character with '%23' so that Matrix is happy
20 Alias.replace(0, 1, "%23");
21
22 const std::string Output = leet::invokeRequest_Get(leet::getAPI("/_matrix/client/v3/directory/room/") + Alias);
23 json reqOutput;
24
25 try {
26 reqOutput = { json::parse(Output) };
27 } catch (const json::parse_error& e) {
28 return "";
29 }
30
31 for (auto& output : reqOutput) {
32 if (output["room_id"].is_string()) {
33 leet::errorCode = 0;
34 return output["room_id"].get<std::string>();
35 } else if (!output["errcode"].is_null()) {
36 leet::errorCode = 1;
37 leet::Error = output["errcode"].get<std::string>();
38 if (output["error"].is_string()) leet::friendlyError = output["error"].get<std::string>();
39
40 return "";
41 }
42 }
43
44 return "";
45}

74 vectorWithVal.push_back(room);
75 }

80/* Returns a single room based on the room ID */
81leet::Room::Room leet::returnRoom(leet::User::CredentialsResponse* resp, leet::Room::Room* room) {
82 using json = nlohmann::json;
83 leet::Room::Room theRoom;
84 json returnOutput;
85
86 try {
87 returnOutput = json::parse(leet::invokeRequest_Get(leet::getAPI("/_matrix/client/v1/rooms/" + room->roomID + "/hierarchy"), resp->accessToken));
88 } catch (const json::parse_error& e) {
89 return theRoom;
90 }
91
92 auto& roomOutput = returnOutput["rooms"];
93
94 for (auto i = roomOutput.begin(); i != roomOutput.end(); ++i) {
95 if (i.value().contains("room_id")) theRoom.roomID = i.value()["room_id"];
96 if (i.value().contains("join_rule")) theRoom.joinRule = i.value()["join_rule"];
97 if (i.value().contains("avatar_url")) theRoom.avatarURL = i.value()["avatar_url"];
98 if (i.value().contains("canonical_alias")) theRoom.Alias = i.value()["canonical_alias"];
99 if (i.value().contains("name")) theRoom.Name = i.value()["name"];
100 if (i.value().contains("num_joined_members")) theRoom.memberCount = i.value()["num_joined_members"];
101 if (i.value().contains("topic")) theRoom.Topic = i.value()["topic"];
102 if (i.value().contains("guest_can_join")) theRoom.guestCanJoin = i.value()["guest_can_join"];
103 if (i.value().contains("world_readable")) theRoom.worldReadable = i.value()["world_readable"];
104 if (i.value().contains("room_type")) theRoom.roomType = i.value()["room_type"];
105 }
106
107 return theRoom;
108}

110leet::Room::Room leet::createRoom(leet::User::CredentialsResponse* resp, leet::Room::RoomConfiguration* conf) {
111 using json = nlohmann::json;
112
113 leet::Room::Room theRoom;
114 json theJson;
115
116 theJson["creation_content"]["m.federate"] = conf->Federate;
117 theJson["name"] = conf->Name;
118 theJson["room_alias_name"] = conf->Alias;
119 theJson["topic"] = conf->Topic;
120 theJson["is_direct"] = conf->directMessage;
121
122 if (conf->Preset == LEET_PRESET_PUBLIC) {
123 theJson["preset"] = "public_chat";
124 } else if (conf->Preset == LEET_PRESET_PRIVATE) {
125 theJson["preset"] = "private_chat";
126 } else if (conf->Preset == LEET_PRESET_TRUSTED_PRIVATE) {
127 theJson["preset"] = "trusted_private_chat";
128 }
129
130 const std::string Output = leet::invokeRequest_Post(leet::getAPI("/_matrix/client/v3/createRoom"), theJson.dump(), resp->accessToken);
131 json reqOutput;
132
133 try {
134 reqOutput = { json::parse(Output) };
135 } catch (const json::parse_error& e) {
136 return theRoom;
137 }
138
139 for (auto& output : reqOutput) {
140 leet::errorCode = 0;
141
142 if (output["room_id"].is_string()) theRoom.roomID = output["room_id"].get<std::string>();
143 if (output["errcode"].is_string()) leet::Error = output["errcode"].get<std::string>();
144 if (output["error"].is_string()) leet::friendlyError = output["error"].get<std::string>();
145 }
146
147 return leet::returnRoom(resp, &theRoom);
148}

215/* Returns a vector of all spaces */
216const std::vector<leet::Space::Space> leet::returnSpaces(leet::User::CredentialsResponse* resp, const int Limit) {
217 std::vector<leet::Space::Space> spaces;
218 std::vector<leet::Room::Room> rooms = leet::returnRoomIDs(resp);
219
220 for (auto& room : rooms) { // each room id
221 leet::Space::Space space;
222 std::vector<leet::Room::Room> roomsInSpace = leet::returnRoomsInSpace(resp, room.roomID, Limit);
223
224 for (auto& roomInSpace : roomsInSpace) { // each room in the space
225 if (roomInSpace.roomType.compare("m.space")) {
226 space.Rooms.push_back(roomInSpace);
227 continue;
228 }
229
230 space.spaceID = roomInSpace.roomID;
231 space.joinRule = roomInSpace.joinRule;
232 space.avatarURL = roomInSpace.avatarURL;
233 space.Alias = roomInSpace.Alias;
234 space.Name = roomInSpace.Name;
235 space.memberCount = roomInSpace.memberCount;
236 space.Topic = roomInSpace.Topic;
237 space.guestCanJoin = roomInSpace.guestCanJoin;
238 space.worldReadable = roomInSpace.worldReadable;
239
240 space.Rooms.push_back(roomInSpace);
241
242 break;
243 }
244
245 spaces.push_back(space);
246 }
247
248 return spaces;
249}
