9/* Converts a username to a proper user ID if necessary (i.e. speedie -> @speedie:matrix.org) */
10const std::string leet::findUserID(const std::string& Alias, const std::string& Homeserver) {
11 if (Alias[0] != '@')
12 return "@" + Alias + ":" + Homeserver;

25/* Returns a leet::User::Profile class containing things such as display name and avatar URL */
26leet::User::Profile leet::getUserData(leet::User::CredentialsResponse* resp, const std::string& userID) {
27 using json = nlohmann::json;
28 leet::errorCode = 0;
29 leet::User::Profile profile;
30
31 if (userID[0] != '@') {
32 return profile;
33 }
34
35 profile.userID = leet::findUserID(userID, resp->Homeserver);
36
37 if (profile.userID.empty()) {
38 leet::errorCode = 1;
39 leet::friendlyError = "Failed to get User ID";
40 return profile;
41 }
42
43 const std::string API { leet::getAPI("/_matrix/client/v3/profile/" + profile.userID) };
44 const std::string Output = invokeRequest_Get(API);
45
46 json reqOutput;
47
48 try {
49 reqOutput = { json::parse(Output) };
50 } catch (const json::parse_error& e) {
51 return profile;
52 }
53
54 for (auto& output : reqOutput) {
55 if (output["avatar_url"].is_string()) profile.avatarURL = output["avatar_url"].get<std::string>();
56 if (output["displayname"].is_string()) profile.displayName = output["displayname"].get<std::string>();
57
58 if (output["errcode"].is_string()) {
59 leet::errorCode = 1;
60 leet::Error = output["errcode"].get<std::string>();
61 if (output["error"].is_string()) leet::friendlyError = output["error"].get<std::string>();
62 }
63 }
64
65 leet::User::Profile userProfile;
66 userProfile.userID = userID;
67 std::vector<leet::User::Profile> User = { userProfile };
68
69 profile.Devices = leet::returnDevicesFromUser(resp, User);
70
71 return profile;
72}

74/* Returns an array of all devices for a user */
75const std::vector<leet::User::Device> leet::returnDevicesFromUser(leet::User::CredentialsResponse* resp, const std::vector<leet::User::Profile>& user) {
76 std::vector<leet::User::Device> devices;
77 using json = nlohmann::json;
78
79 json Body;
80 json deviceKeys;
81
82 for (auto& theUser : user) {
83 deviceKeys[theUser.userID] = json::array();
84 Body["device_keys"] = deviceKeys;
85 Body["timeout"] = 10000;
86 }
87
88 const std::string Output = leet::invokeRequest_Post(leet::getAPI("/_matrix/client/v3/keys/query"), Body.dump(), resp->accessToken);
89 json returnOutput;
90
91 try {
92 returnOutput = json::parse(Output);
93 } catch (const json::parse_error& e) {
94 return devices;
95 }
96
97 for (auto& userID : user) {
98 auto& deviceList = returnOutput["device_keys"][userID.userID];
99
100 for (auto it = deviceList.begin(); it != deviceList.end(); ++it) {
101 leet::User::Device device;
102
103 device.userID = userID.userID;
104 device.deviceID = it.key();
105
106 json::json_pointer curve25519Pointer("/keys/curve25519:" + device.deviceID);
107 if (it.value().contains(curve25519Pointer)) {
108 device.curve25519Key = it.value()["keys"]["curve25519:" + device.deviceID];
109 }
110 json::json_pointer ed25519Pointer("/keys/ed25519:" + device.deviceID);
111 if (it.value().contains(ed25519Pointer)) {
112 device.ed25519Key = it.value()["keys"]["ed25519:" + device.deviceID];
113 }
114 json::json_pointer ed25519SigPointer("/signatures/" + userID.userID + "/ed25519:" + device.deviceID);
115 if (it.value().contains(ed25519SigPointer)) {
116 device.ed25519Signature = it.value()["signatures"][userID.userID]["ed25519:" + device.deviceID];
117 }
118 if (it.value().contains("/unsigned/device_display_name"_json_pointer)) {
119 device.deviceDisplayName = it.value()["unsigned"]["device_display_name"];
120 }
121
122 device.olm = false;
123 device.megolm = false;
124
125 if (std::find(it.value()["algorithms"].begin(), it.value()["algorithms"].end(), "m.olm.v1.curve25519-aes-sha2") != it.value()["algorithms"].end())
126 device.olm = true;
127
128 if (std::find(it.value()["algorithms"].begin(), it.value()["algorithms"].end(), "m.megolm.v1.aes-sha2") != it.value()["algorithms"].end())
129 device.megolm = true;
130
131 devices.push_back(device);
132 }
133 }
134
135 return devices;
136}

138/* Returns true if username is valid and available for registering */
139const bool leet::checkIfUsernameIsAvailable(const std::string& Username) {
140 using json = nlohmann::json;
141 leet::errorCode = 0;
142
143 std::string theUsername = Username;
144
145 if (Username[0] == '@') {
146 theUsername = leet::returnUserName(Username);
147
148 if (theUsername[0] == '@' || !theUsername.compare("")) {
149 return false;
150 leet::errorCode = 1;
151 }
152 }
153
154 const std::string API { leet::getAPI("/_matrix/client/v3/register/available?username=" + theUsername) };
155 const std::string Output = invokeRequest_Get(API);
156
157 json reqOutput;
158
159 try {
160 reqOutput = { json::parse(Output) };
161 } catch (const json::parse_error& e) {
162 return false; // fallback to false is probably safest?
163 }
164
165 for (auto& output : reqOutput) {
166 if (output["errcode"].is_string()) {
167 leet::errorCode = 1;
168 leet::Error = output["errcode"].get<std::string>();
169 if (output["error"].is_string()) leet::friendlyError = output["error"].get<std::string>();
170 return false;
171 }
172
173 if (output["available"].is_boolean()) {
174 return output["available"].get<bool>();
175 }
176 }
177
178 return false;
179}

181/* Returns an array of all users in a room */
182const std::vector<leet::User::Profile> leet::returnUsersInRoom(leet::User::CredentialsResponse* resp, leet::Room::Room* room) {
183 using json = nlohmann::json;
184 std::vector<leet::User::Profile> vector;
185
186 const std::string Output = leet::invokeRequest_Get(leet::getAPI("/_matrix/client/v3/rooms/" + room->roomID + "/joined_members"), resp->accessToken);
187 json returnOutput;
188
189 try {
190 returnOutput = json::parse(Output);
191 } catch (const json::parse_error& e) {
192 return vector;
193 }
194
195 auto& users = returnOutput["joined"];
196
197 for (auto it = users.begin(); it != users.end(); ++it) {
198 leet::User::Profile profile;
199
200 if (it.value().contains("avatar_url") && !it.value()["avatar_url"].is_null()) profile.avatarURL = it.value()["avatar_url"];
201 if (it.value().contains("display_name") && !it.value()["display_name"].is_null()) profile.displayName = it.value()["display_name"];
202
203 profile.userID = it.key();
204
205 leet::User::Profile userProfile;
206 userProfile.userID = it.key();
207 std::vector<leet::User::Profile> User = { userProfile };
208 profile.Devices = leet::returnDevicesFromUser(resp, User);
209
210 vector.push_back(profile);
211 }
212
213 return vector;
214}
